class SheepFarmModel {
  constructor(inputs) {
    this.inputs = inputs;
  }
  
  calculate() {
    const cc = TABLES.carryingCapacity[this.inputs.breed][this.inputs.productionSystem];
    const maxStock = this.inputs.landHa / cc;
    const isOverstocked = this.inputs.flockSize > maxStock;
    
    const repro = TABLES.reproduction[this.inputs.productionSystem];
    const lambsWeaned = Math.floor(this.inputs.flockSize * repro.weaningRate * (1 - repro.preWeaningMortality));
    const lambsSold = Math.floor(lambsWeaned * (1 - repro.postWeaningMortality));
    
    let meatPrice = TABLES.meatPrices.lamb;
    if (this.inputs.marketChannel === 'direct') meatPrice *= 1.20;
    else if (this.inputs.marketChannel === 'abattoir') meatPrice *= 1.05;
    const meatRevenue = lambsSold * TABLES.meatPrices.carcassWeight * meatPrice;
    
    const woolProd = TABLES.woolProduction[this.inputs.breed];
    const woolRevenue = woolProd.avg > 0 ? this.inputs.flockSize * woolProd.avg * woolProd.price : 0;
    
    let otherRevenue = this.inputs.flockSize * TABLES.otherRevenue.manurePerEwe;
    otherRevenue += lambsSold * TABLES.otherRevenue.hidePerLamb;
    if (this.inputs.studOperation) otherRevenue += this.inputs.flockSize * TABLES.otherRevenue.studPremiumPerEwe;
    if (this.inputs.agriTourism) otherRevenue += TABLES.otherRevenue.agriTourismAnnual;
    
    const varCosts = TABLES.variableCosts[this.inputs.productionSystem];
    let feed = varCosts.feedPasture + varCosts.feedConcentrate;
    if (this.inputs.feedSource === 'homeGrown') feed *= 0.70;
    if (this.inputs.feedSource === 'mixed') feed *= 0.85;
    const variableCosts = (feed + varCosts.vet + varCosts.shearing + varCosts.transport + varCosts.waterUtilities) * this.inputs.flockSize;
    
    let workers = 1;
    if (this.inputs.productionSystem === 'extensive') workers = Math.max(1, Math.ceil(this.inputs.flockSize / 200));
    else if (this.inputs.productionSystem === 'semiIntensive') workers = Math.max(1, Math.ceil(this.inputs.flockSize / 150));
    else workers = Math.max(1, Math.ceil(this.inputs.flockSize / 100));
    const labourCosts = workers * 8 * 365 * TABLES.labour.typicalSkilled;
    
    const perimeterM = Math.sqrt(this.inputs.landHa) * 4 * 100;
    const fencing = (perimeterM * TABLES.fixedCosts.fencingPerMeter) / 10;
    const fixedCosts = fencing + TABLES.fixedCosts.handlingFacilities + TABLES.fixedCosts.shearingShed + TABLES.fixedCosts.waterInfra;
    
    const totalCosts = variableCosts + labourCosts + fixedCosts;
    const totalRevenue = meatRevenue + woolRevenue + otherRevenue;
    const netProfit = totalRevenue - totalCosts;
    const profitMargin = totalRevenue > 0 ? (netProfit / totalRevenue) * 100 : 0;
    const profitPerLamb = (meatRevenue + (woolRevenue / Math.max(1, lambsSold))) - (variableCosts / Math.max(1, lambsSold));
    const breakEvenLambs = profitPerLamb > 0 ? Math.ceil(totalCosts / profitPerLamb) : 999;
    
    return {
      netProfit, profitMargin, totalRevenue, totalCosts,
      lambsWeaned, lambsSold, maxStock, isOverstocked,
      profitPerEwe: netProfit / this.inputs.flockSize,
      breakEvenLambs, meatRevenue, woolRevenue, otherRevenue,
      variableCosts, labourCosts, fixedCosts,
      carryingCapacityRate: cc, stockingDeficit: isOverstocked ? this.inputs.flockSize - maxStock : 0
    };
  }
  
  getStupidityReport() {
    return Object.entries(TABLES.stupidityBaselines).map(([key, val]) => ({
      component: key, si: val.si, action: val.action, phone: val.phone, saving: val.saving
    }));
  }
  
  getRecommendations(results) {
    const week = [], month = [], season = [];
    const stupidity = this.getStupidityReport();
    
    stupidity.filter(s => s.si > 5).forEach(s => week.push(`${s.action} → Save ${s.saving}`));
    if (results.isOverstocked) month.push(`Reduce flock to ${Math.floor(results.maxStock)} ewes`);
    if (this.inputs.marketChannel === 'auction') season.push(`Switch to direct sales → +20% margin`);
    if (this.inputs.feedSource === 'purchased') month.push(`Buy feed in bulk → save 15%`);
    
    return { week, month, season };
  }
}

module.exports = SheepFarmModel;

class InputSensitivityAnalyzer {
  constructor() {
    this.defaults = {
      landHa: 100, flockSize: 150, breed: "Merino",
      productionSystem: "semiIntensive", marketChannel: "auction",
      feedSource: "mixed", studOperation: false, agriTourism: false
    };
  }
  
  analyze(model) {
    const impacts = [];
    const baseProfit = model.netProfit;
    
    const tests = [
      { key: "flockSize", delta: 0.2 },
      { key: "landHa", delta: 0.2 },
      { key: "marketChannel", alternatives: ["direct", "abattoir"] }
    ];
    
    for (let test of tests) {
      if (test.delta) {
        const newInputs = { ...this.defaults, [test.key]: this.defaults[test.key] * (1 + test.delta) };
        const newModel = new SheepFarmModel(newInputs);
        newModel.calculate();
        const impact = Math.abs((newModel.results.netProfit - baseProfit) / baseProfit) * 100;
        impacts.push({ field: test.key, impactPercent: impact, priority: impact > 5 ? "HIGH" : "MEDIUM" });
      }
    }
    return impacts.sort((a,b) => b.impactPercent - a.impactPercent);
  }
  
  getHighImpactFields() {
    return [
      { field: "flockSize", label: "Number of ewes", reason: "Directly scales revenue and costs", priority: "HIGH" },
      { field: "feedSource", label: "Feed source", reason: "60-70% of variable costs", priority: "HIGH" },
      { field: "marketChannel", label: "Market channel", reason: "Direct adds 20% margin", priority: "HIGH" }
    ];
  }
}

async function requireSession(req, res, next) {
  const sessionId = req.headers['x-session-id'];
  if (!sessionId) return res.status(401).json({ error: 'No session' });
  db.get('SELECT user_id, expires FROM sessions WHERE id = ?', [sessionId], (err, row) => {
    if (err || !row || row.expires < Date.now()) return res.status(401).json({ error: 'Invalid session' });
    req.userId = row.user_id;
    next();
  });
}

  sql += ' ORDER BY created_at DESC LIMIT 50';
  db.all(sql, params, (err, rows) => {
    res.json(rows || []);
  });
});

PORT=3000

sudo tee /etc/nginx/sites-available/shepherdai << 'NGINX'
server {
    listen 80;

        proxy_pass http://127.0.0.1:3000;
