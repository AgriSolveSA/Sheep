Fraud Prevention Code Snippet

```javascript
// Server-side validation for new listings
function validateNewListing(userId, listingData) {
  // Rate limiting: max 5 ads per day per user
  const recentCount = db.get(
    "SELECT COUNT(*) FROM listings WHERE user_id = ? AND created_at > datetime('now', '-1 day')",
    [userId]
  );
  if (recentCount >= 5) {
    throw new Error("Daily listing limit reached");
  }

  // Duplicate detection: similar title + description within 30 days
  const duplicate = db.get(
    "SELECT 1 FROM listings WHERE user_id = ? AND title LIKE ? AND created_at > datetime('now', '-30 days')",
    [userId, `%${listingData.title}%`]
  );
  if (duplicate) {
    throw new Error("Similar listing already exists. Please renew existing ad.");
  }

  // Auto-moderation: check for blacklisted keywords
  const blacklisted = ['scam', 'fake', 'counterfeit'];
  if (blacklisted.some(keyword => listingData.description.toLowerCase().includes(keyword))) {
    return { status: 'flagged', requires_review: true };
  }

  return { status: 'approved' };
}
```
